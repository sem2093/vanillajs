JavaScript / Loops
Loops / For

Ex: let str = ''; for (let i = 0; i < 9; i++) { str = str + i; } console.log(str);

for (let i = 0; i < 4; i += 1) { console.log(i); }
Loops / For...In

Ex:const object = { a: 1, b: 2, c: 3 }; for (const property in object) { console.log(`${property}: ${object[property]}`); } // Expected output: // "a: 1" // "b: 2" // "c: 3"

const obj = { a: 15, b: 35, c: 45 }; for (const prop in obj) { console.log(`obj.${prop} = ${obj[prop]}`); } // Logs: // "obj.a = 15" // "obj.b = 35" // "obj.c = 45"
Loops / While

Ex: let n = 0; while (n < 3) { n++; } console.log(n);

while (i < 10) { text += "The number is " + i; i++; }
Loops / Do...While

Ex:let result = ''; let i = 0; do { i = i + 1; result = result + i; } while (i < 5); console.log(result); // Expected output: "12345"

let text = "These Docs Rock"; let i = 0; do { text += i + "
"; i++; } while (i < 5);
Loops / For...Of

Ex: // string const string = 'code'; // using for...of loop for (let i of string) { console.log(i); }

const array1 = ['a', 'b', 'c']; for (const element of array1) { console.log(element); } // Expected output: "a" // Expected output: "b" // Expected output: "c"
Loops / For Await...Of

Ex: async function* asyncGenerator() { let i = 0; while (i < 3) yield i++; } } (async () => { for await (const num of asyncGenerator()) { console.log(num); } })(); // 0 // 1 // 2
